## 📖 **Usage Examples**

### **Background Jobs**

```typescript
import { 
  initializeBackgroundJobs,
  scheduleGoalReminder, 
  scheduleBudgetAlert,
  getJobQueueStats
} from './services/backgroundJobs'

// Initialize on app start
initializeBackgroundJobs()

// Schedule goal reminder
scheduleGoalReminder(
  1, 
  'Save £10k', 
  new Date('2026-12-31')
)

// Schedule budget alert (90% of budget used)
scheduleBudgetAlert('food', 450, 500)

// Get job queue statistics
const stats = getJobQueueStats()
console.log(stats) // { total, pending, running, completed, failed }
```

### **Form Validation**

```typescript
import { validators } from './hooks/useAdvancedValidation'
import { useForm } from './utils/formValidation'

const form = useForm({
  initialValues: {
    sortCode: '',
    accountNumber: '',
    amount: ''
  },
  validationConfig: {
    sortCode: { 
      rules: [validators.ukSortCode()] 
    },
    accountNumber: { 
      rules: [validators.ukAccountNumber()] 
    },
    amount: { 
      rules: [
        validators.positiveAmount(),
        validators.maxAmount(10000)
      ] 
    }
  },
  onSubmit: async (values) => {
    // Form is valid, handle submission
    console.log('Valid data:', values)
  }
})
```
