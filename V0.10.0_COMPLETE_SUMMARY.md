## 🎓 Usage Guide

### API Client
```typescript
import { createApiClient, createAuthInterceptor } from '@/utils/apiClient'

const api = createApiClient({
  baseURL: 'https://api.example.com',
  timeout: 30000,
  cache: true,
})

api.addRequestInterceptor(createAuthInterceptor(() => token))
const data = await api.get('/endpoint')
```

### State Management
```typescript
import { Store, useStore, useSelector } from '@/utils/stateManagement'

const store = new Store(initialState, reducer, middlewares)

function Component() {
  const value = useSelector(store, (state) => state.value)
  return <div>{value}</div>
}
```

### Form Validation
```typescript
import { useForm, required, email } from '@/utils/formValidation'

const form = useForm({
  initialValues: { email: '' },
  validationConfig: {
    email: { rules: [required(), email()] },
  },
  onSubmit: (values) => console.log(values),
})
```
